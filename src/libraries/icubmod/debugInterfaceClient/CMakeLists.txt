PREPARE_DEVICE(debugInterfaceClient TYPE DebugInterfaceClient INCLUDE debugInterfaceClient.h)

if(NOT SKIP_debugInterfaceClient)

			${iCubDev_INCLUDE_DIRS}
                       ${YARP_INCLUDE_DIRS})

icub_export_library(debugInterfaceClient)
yarp_install(FILES debugInterfaceClient_manifest.ini  DESTINATION ${ICUB_PLUGIN_MANIFESTS_INSTALL_DIR})

endif(NOT SKIP_debugInterfaceClient)
